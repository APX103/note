1架构总览

Agent 与平台的双向关系

sequenceDiagram autonumber participant Agent as Agent 进程
(你的代码) participant SDK as Agent SDK
(内置 HTTP Client) participant Platform as A2A 平台
(NestJS) participant LLM as LLM API
(OpenAI/Claude) Note over Agent,Platform: 启动阶段 Agent->>SDK: new A2AAgentSDK(config) SDK->>Platform: POST /registry/register {AgentCard} Platform-->>SDK: { id, token } SDK->>SDK: 启动心跳定时器 (15s) Note over Agent,Platform: 运行阶段 loop 每 15s SDK->>Platform: POST /registry/heartbeat end Note over Agent,Platform: 被调用时（平台 → Agent） Platform->>Agent: POST /sendMessage {task, payload, session} Agent->>Agent: 解析请求 Agent->>LLM: 发送 prompt + session context LLM-->>Agent: 返回结果 Agent-->>Platform: { body, artifact?, status } Note over Agent,Platform: 主动调用时（Agent → 平台 → 其他 Agent） Agent->>SDK: sdk.sendMessage(toAgentId, payload, taskId) SDK->>Platform: POST /sendMessage + x-parent-message-id Platform->>Platform: 路由到目标 Agent Platform-->>SDK: 目标 Agent 的回复 SDK-->>Agent: 返回结果

Agent 端需要实现 两个角色：

角色职责对应代码
A2A Server对外暴露 HTTP 端点，接收平台（或其他 Agent）发来的消息Express / Fastify / NestJS HTTP Server
A2A Client通过 SDK 调用平台接口，发现其他 Agent、发送消息、创建 TaskA2AAgentSDK 类

核心原则：Agent 不直接与其他 Agent 通信，所有消息都经过平台转发。Agent 的 HTTP Server 只接收来自平台的请求，Agent 的 SDK 只向平台发送请求。

2平台 API 完整规范

所有 API 的 Base URL：http://<platform-host>:3000，请求头需携带 Authorization: Bearer <agent-jwt-token>

§ Registry API（服务发现）

POST /registry/register

Agent 启动时调用，注册自己到平台。如果已存在相同 endpointUrl，则更新。

// Request
{
  "name": "weather-agent",
  "description": "提供全球城市天气查询",
  "endpointUrl": "http://weather.local:8001",   // Agent 自己的 HTTP 地址
  "version": "1.0.0",
  "capabilities": { "streaming": true, "pushNotifications": false },
  "skills": [
    { "id": "get_current_weather", "name": "当前天气", "description": "查询指定城市当前天气" },
    { "id": "get_forecast", "name": "天气预报", "description": "查询未来7天天气预报" }
  ],
  "securitySchemes": { "type": "bearer" }
}

// Response 200
{
  "id": "agent_weather_001",
  "name": "weather-agent",
  "endpointUrl": "http://weather.local:8001",
  "status": "ACTIVE",
  "createdAt": "2025-05-09T10:00:00Z"
}

每 15 秒调用一次，证明 Agent 还活着。

§ Gateway API（消息通信）

GET /.well-known/agent-card.json?agentId=<id>

获取指定 Agent 的公开 Agent Card（A2A 标准端点）。

POST /sendMessage

平台代理 API：同步发送消息。Agent 通过此接口调用其他 Agent（经过平台转发）。注：A2A 官方标准为 JSON-RPC POST /，method 为 SendMessage。

// Request Headers
Content-Type: application/json
Authorization: Bearer <agent-jwt-token>
x-trace-id: <optional>                    // 透传 Trace ID
x-parent-message-id: <optional>           // 父消息 ID，建立 DAG 调用树

// Request Body
{
  "fromAgentId": "agent_001",              // 发送方 Agent ID
  "toAgentId": "agent_002",                // 目标 Agent ID
  "taskId": "task_abc_123",                // 所属 Task ID（可选）
  "messageType": "TASK_REQUEST",           // TEXT / TASK_REQUEST / TASK_RESPONSE / STATUS_UPDATE
  "payload": {
    "body": "请查询北京明天天气",
    "input": { "city": "北京", "date": "2025-05-10" }
  }
}

// Response 200 — 目标 Agent 的回复（平台透传）
{
  "body": "北京明天晴，25°C，适合出行",
  "artifact": {
    "type": "weather_data",
    "data": { "city": "北京", "date": "2025-05-10", "temp": 25, "condition": "晴" }
  }
}

POST /sendMessageStream

平台代理 API：SSE 流式发送消息。用于长任务实时推送进度。注：A2A 官方标准为 SendStreamingMessage。

// Request Body（同 /sendMessage）

// Response — text/event-stream
id: 1
event: task_status
data: {"status": "WORKING", "progress": 30}

id: 2
event: artifact
data: {"type": "partial_result", "content": "..."}

id: 3
event: task_status
data: {"status": "COMPLETED", "result": "..."}

3Agent SDK 完整实现

Agent SDK 是一个 TypeScript 类，封装了所有平台 API 调用。Agent 开发者只需实例化 SDK，即可自动完成注册、心跳、消息收发。

// agent-sdk/src/types.ts
// ───────────────────────────────────────────────────────────────

export interface AgentCard {

  description: string;

    streaming?: boolean;

// agent-sdk/src/a2a-agent-sdk.ts
// ───────────────────────────────────────────────────────────────

import { AgentCard, SendMessageRequest, SendMessageResponse, TaskCreateRequest, TaskResponse, SessionContext } from './types';

export interface SDKConfig {
  platformUrl: string;        // 平台地址，如 http://localhost:3000
  agentCard: AgentCard;       // Agent 自我介绍
  token: string;              // JWT Token（从平台获取或预分配）
  heartbeatInterval?: number;  // 心跳间隔（默认 15000ms）
  pollInterval?: number;       // 轮询间隔（默认 5000ms）
  deliveryStrategy?: 'polling' | 'websocket' | 'callback'; // 消息投递策略（默认 polling）
}

export class A2AAgentSDK {
  private config: SDKConfig;
  private agentId: string | null = null;
  private heartbeatTimer: NodeJS.Timer | null = null;
  private pollTimer: NodeJS.Timer | null = null;
  private messageHandler: ((message: any) => Promise<any>) | null = null;

  constructor(config: SDKConfig) {
    this.config = {
      heartbeatInterval: 15000,
      pollInterval: 5000,
      deliveryStrategy: 'polling',
      ...config,
    };
  }

  // ═══════════════════════════════════════════════════════════════
  // 生命周期管理
  // ═══════════════════════════════════════════════════════════════

  /**
   * 启动 Agent：注册到平台 + 启动心跳 + 启动轮询（默认策略）
   */
  async start(): Promise<void> {
    const registered = await this.register();
    this.agentId = registered.id;
    this.startHeartbeat();

    if (this.config.deliveryStrategy === 'polling') {
      this.startPolling();
    }

    console.log(`[A2A SDK] Agent started: ${this.agentId}, strategy: ${this.config.deliveryStrategy}`);
  }

  /**
   * 停止 Agent：注销 + 停止心跳 + 停止轮询
   */
  async stop(): Promise<void> {
    this.stopPolling();
    this.stopHeartbeat();
    if (this.agentId) {
      await this.unregister();
      console.log(`[A2A SDK] Agent stopped: ${this.agentId}`);
    }
  }

  // ═══════════════════════════════════════════════════════════════
  // 轮询逻辑（默认策略）
  // ═══════════════════════════════════════════════════════════════

  private startPolling(): void {
    this.pollTimer = setInterval(() => {
      this.pollInbox().catch(console.error);
    }, this.config.pollInterval);
  }

  private stopPolling(): void {
    if (this.pollTimer) {
      clearInterval(this.pollTimer);
      this.pollTimer = null;
    }
  }

  private async pollInbox(): Promise<void> {
    if (!this.agentId) return;

    const res = await fetch(
      `${this.config.platformUrl}/inbox/poll?agentId=${this.agentId}&limit=10`,
      { headers: { 'Authorization': `Bearer ${this.config.token}` } }
    );
    if (!res.ok) return;

    const messages = await res.json();
    for (const msg of messages) {
      console.log(`[A2A SDK] Poll received from ${msg.fromAgentId}: ${msg.body}`);

      if (this.messageHandler) {
        const result = await this.messageHandler(msg);
        // 发送回复
        if (msg.fromAgentId && result) {
          await this.sendReply(msg.fromAgentId, result, msg.taskId, msg.messageId);
        }
      }
    }
  }

  private async sendReply(
    toAgentId: string,
    result: any,
    taskId?: string,
    parentMessageId?: string,
  ): Promise<void> {
    await fetch(`${this.config.platformUrl}/sendMessage`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.config.token}`,
      },
      body: JSON.stringify({
        fromAgentId: this.agentId,
        toAgentId,
        taskId,
        parentMessageId,
        messageType: 'TASK_RESPONSE',
        payload: result,
      }),
    });
  }

  /**
   * 注册消息处理器（Agent 开发者调用）
   * 在 polling 策略下，SDK 轮询到消息后会调用此 handler
   */
  onMessage(handler: (message: any) => Promise<any>): void {
    this.messageHandler = handler;
  }

  // ═══════════════════════════════════════════════════════════════
  // Registry API 封装
  // ═══════════════════════════════════════════════════════════════

  /** 注册 Agent */
  private async register(): Promise<{ id: string }> {
    const res = await fetch(`${this.config.platformUrl}/registry/register`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.config.token}`,
      },
      body: JSON.stringify(this.config.agentCard),
    });
    if (!res.ok) throw new Error(`Register failed: ${res.status} ${await res.text()}`);
    return res.json();
  }

  /** 注销 Agent */
  private async unregister(): Promise<void> {
    const res = await fetch(`${this.config.platformUrl}/registry/agents/${this.agentId}`, {
      method: 'DELETE',
      headers: { 'Authorization': `Bearer ${this.config.token}` },
    });
    if (!res.ok) console.warn(`Unregister warning: ${res.status}`);
  }

  /** 发送心跳 */
  private async sendHeartbeat(): Promise<void> {
    if (!this.agentId) return;
    const res = await fetch(`${this.config.platformUrl}/registry/heartbeat`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.config.token}`,
      },
      body: JSON.stringify({ agentId: this.agentId }),
    });
    if (!res.ok) console.warn(`Heartbeat failed: ${res.status}`);
  }

  private startHeartbeat(): void {
    this.heartbeatTimer = setInterval(() => {
      this.sendHeartbeat().catch(console.error);
    }, this.config.heartbeatInterval);
  }

  private stopHeartbeat(): void {
    if (this.heartbeatTimer) {
      clearInterval(this.heartbeatTimer);
      this.heartbeatTimer = null;
    }
  }

  // ═══════════════════════════════════════════════════════════════
  // Discovery API 封装
  // ═══════════════════════════════════════════════════════════════

  /**
   * 发现具备指定技能的 Agent
   */
  async discoverAgents(skill: string): Promise<AgentCard[]> {
    const url = new URL(`${this.config.platformUrl}/registry/agents`);
    url.searchParams.set('skill', skill);
    const res = await fetch(url.toString(), {
      headers: { 'Authorization': `Bearer ${this.config.token}` },
    });
    if (!res.ok) throw new Error(`Discover failed: ${res.status}`);
    return res.json();
  }

  /**
   * 关键词搜索 Agent
   */
  async searchAgents(query: string): Promise<AgentCard[]> {
    const url = new URL(`${this.config.platformUrl}/registry/agents`);
    url.searchParams.set('q', query);
    const res = await fetch(url.toString(), {
      headers: { 'Authorization': `Bearer ${this.config.token}` },
    });
    if (!res.ok) throw new Error(`Search failed: ${res.status}`);
    return res.json();
  }

  /**
   * 获取指定 Agent 的 Agent Card
   */
  async getAgentCard(agentId: string): Promise<AgentCard> {
    const res = await fetch(`${this.config.platformUrl}/registry/agents/${agentId}`, {
      headers: { 'Authorization': `Bearer ${this.config.token}` },
    });
    if (!res.ok) throw new Error(`Get agent card failed: ${res.status}`);
    return res.json();
  }

  // ═══════════════════════════════════════════════════════════════
  // Gateway API 封装（发送消息）
  // ═══════════════════════════════════════════════════════════════

  /**
   * 发送同步消息（通过平台转发到目标 Agent）
   * @param toAgentId 目标 Agent ID
   * @param payload 消息内容
   * @param taskId 所属 Task（可选）
   * @param parentMessageId 父消息 ID（建立 DAG 调用树，可选）
   */
  async sendMessage(
    toAgentId: string,
    payload: SendMessagePayload,
    taskId?: string,
    parentMessageId?: string,
  ): Promise<SendMessageResponse> {
    const body: SendMessageRequest = {
      fromAgentId: this.agentId!,
      toAgentId,
      taskId,
      messageType: taskId ? 'TASK_REQUEST' : 'TEXT',
      payload,
    };

    const headers: Record<string, string> = {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${this.config.token}`,
    };
    if (parentMessageId) {
      headers['x-parent-message-id'] = parentMessageId;
    }

    // 发送消息到平台（存入对方 Inbox）
    const res = await fetch(`${this.config.platformUrl}/sendMessage`, {
      method: 'POST',
      headers,
      body: JSON.stringify(body),
    });
    if (!res.ok) throw new Error(`Send message failed: ${res.status} ${await res.text()}`);

    const sent = await res.json();  // { messageId, status: 'queued' }

    // 轮询模式下：封装轮询等待，对开发者保持同步语义
    if (this.config.deliveryStrategy === 'polling') {
      return this.pollForReply(sent.messageId, { timeout: 30000, interval: 2000 });
    }

    return sent;
  }

  /**
   * 轮询等待指定消息的回复（polling 策略下使用）
   */
  private async pollForReply(
    sentMessageId: string,
    options: { timeout: number; interval: number },
  ): Promise<SendMessageResponse> {
    const deadline = Date.now() + options.timeout;

    while (Date.now() < deadline) {
      const res = await fetch(
        `${this.config.platformUrl}/inbox/poll?agentId=${this.agentId}&limit=10`,
        { headers: { 'Authorization': `Bearer ${this.config.token}` } }
      );
      if (!res.ok) { await sleep(options.interval); continue; }

      const messages = await res.json();
      for (const msg of messages) {
        if (msg.parentMessageId === sentMessageId) {
          return { body: msg.body, artifact: msg.artifact };
        }
      }

      await sleep(options.interval);
    }

    throw new Error(`Reply timeout for message ${sentMessageId}`);
  }

  /**
   * 发送 SSE 流式消息
   */
  async sendMessageStream(
    toAgentId: string,
    payload: SendMessagePayload,
    taskId?: string,
    onEvent?: (event: any) => void,
  ): Promise<void> {
    const body: SendMessageRequest = {
      fromAgentId: this.agentId!,
      toAgentId,
      taskId,
      messageType: 'TASK_REQUEST',
      payload,
    };

    const res = await fetch(`${this.config.platformUrl}/sendMessageStream`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.config.token}`,
      },
      body: JSON.stringify(body),
    });

    if (!res.ok) throw new Error(`Stream failed: ${res.status}`);
    if (!res.body) return;

    const reader = res.body.getReader();
    const decoder = new TextDecoder();
    let buffer = '';

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      buffer += decoder.decode(value, { stream: true });

      const lines = buffer.split('\n');
      buffer = lines.pop() || '';

      for (const line of lines) {
        if (line.startsWith('data: ')) {
          const data = line.slice(6);
          try {
            onEvent?.(JSON.parse(data));
          } catch {
            onEvent?.(data);
          }
        }
      }
    }
  }

  // ═══════════════════════════════════════════════════════════════
  // Task API 封装
  // ═══════════════════════════════════════════════════════════════

  /**
   * 创建 Task
   */
  async createTask(request: TaskCreateRequest): Promise<TaskResponse> {
    const res = await fetch(`${this.config.platformUrl}/tasks`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.config.token}`,
      },
      body: JSON.stringify(request),
    });
    if (!res.ok) throw new Error(`Create task failed: ${res.status}`);
    return res.json();
  }

  /** 获取 Task 详情 */
  async getTask(taskId: string): Promise<TaskResponse> {
    const res = await fetch(`${this.config.platformUrl}/tasks/${taskId}`, {
      headers: { 'Authorization': `Bearer ${this.config.token}` },
    });
    if (!res.ok) throw new Error(`Get task failed: ${res.status}`);
    return res.json();
  }

  /** 标记 Task 完成 */
  async completeTask(taskId: string): Promise<TaskResponse> {
    const res = await fetch(`${this.config.platformUrl}/tasks/${taskId}/complete`, {
      method: 'POST',
      headers: { 'Authorization': `Bearer ${this.config.token}` },
    });
    if (!res.ok) throw new Error(`Complete task failed: ${res.status}`);
    return res.json();
  }

  /** 请求人工介入 */
  async escalateTask(taskId: string, reason: string): Promise<TaskResponse> {
    const res = await fetch(`${this.config.platformUrl}/tasks/${taskId}/escalate`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.config.token}`,
      },
      body: JSON.stringify({ reason }),
    });
    if (!res.ok) throw new Error(`Escalate task failed: ${res.status}`);
    return res.json();
  }

  // ═══════════════════════════════════════════════════════════════
  // Session API 封装
  // ═══════════════════════════════════════════════════════════════

  /** 获取 Session Context */
  async getSession(taskId: string): Promise<SessionContext> {
    const res = await fetch(`${this.config.platformUrl}/session/${taskId}`, {
      headers: { 'Authorization': `Bearer ${this.config.token}` },
    });
    if (!res.ok) throw new Error(`Get session failed: ${res.status}`);
    return res.json();
  }

  // ═══════════════════════════════════════════════════════════════
  // 工具注入（供 LLM 调用）
  // ═══════════════════════════════════════════════════════════════

  /**
   * 生成 discover_agents Tool，注入到 LLM 可用工具集中
   */
  getDiscoverTool() {
    const sdk = this;
    return {
      name: 'discover_agents',
      description: '发现平台中具备特定技能的 Agent。当你需要其他 Agent 协助完成任务时调用此工具。',
      parameters: {
        type: 'object',
        properties: {
          skill: {
            type: 'string',
            description: '要搜索的技能 ID，如 "get_current_weather"、"search_flights"',
          },
          query: {
            type: 'string',
            description: '自然语言描述需求，用于模糊搜索',
          },
        },
        required: ['skill'],
      },
      execute: async (params: { skill: string; query?: string }) => {
        if (params.query) {
          return sdk.searchAgents(params.query);
        }
        return sdk.discoverAgents(params.skill);
      },
    };
  }

  /**
   * 生成 send_message Tool，注入到 LLM 可用工具集中
   */
  getSendMessageTool() {
    const sdk = this;
    return {
      name: 'send_message_to_agent',
      description: '向指定 Agent 发送消息请求协作。调用前应先使用 discover_agents 找到目标 Agent。',
      parameters: {
        type: 'object',
        properties: {
          toAgentId: {
            type: 'string',
            description: '目标 Agent ID',
          },
          body: {
            type: 'string',
            description: '消息内容',
          },
          taskId: {
            type: 'string',
            description: '所属 Task ID（如果在一个已有 Task 中协作）',
          },
          input: {
            type: 'object',
            description: '结构化输入参数',
          },
        },
        required: ['toAgentId', 'body'],
      },
      execute: async (params: { toAgentId: string; body: string; taskId?: string; input?: any }) => {
        return sdk.sendMessage(params.toAgentId, { body: params.body, input: params.input }, params.taskId);
      },
    };
  }

  // ═══════════════════════════════════════════════════════════════
  // Getter
  // ═══════════════════════════════════════════════════════════════

  getAgentId(): string | null {
    return this.agentId;
  }
}

function sleep(ms: number): Promise<void> {
  return new Promise(resolve => setTimeout(resolve, ms));
}

4Agent A2A Server 实现

Agent 需要暴露一个 HTTP Server，接收平台转发来的消息。以下是 Express 实现（最轻量）：

// agent-server/src/index.ts
// ───────────────────────────────────────────────────────────────

import express, { Request, Response } from 'express';
import { A2AAgentSDK } from '@your-org/a2a-sdk';
import { OpenAI } from 'openai';

const app = express();
app.use(express.json());

// ═══════════════════════════════════════════════════════════════
// 配置
// ═══════════════════════════════════════════════════════════════

const PLATFORM_URL = process.env.PLATFORM_URL || 'http://localhost:3000';
const AGENT_TOKEN = process.env.AGENT_TOKEN!;
const OPENAI_API_KEY = process.env.OPENAI_API_KEY!;
const PORT = parseInt(process.env.PORT || '8001');

const openai = new OpenAI({ apiKey: OPENAI_API_KEY });

// ═══════════════════════════════════════════════════════════════
// 初始化 SDK
// ═══════════════════════════════════════════════════════════════

const sdk = new A2AAgentSDK({
  platformUrl: PLATFORM_URL,
  token: AGENT_TOKEN,
  agentCard: {
    name: 'weather-agent',
    description: '全球城市天气查询 Agent',
    endpointUrl: `http://localhost:${PORT}`,
    version: '1.0.0',
    capabilities: { streaming: false },
    skills: [
      { id: 'get_current_weather', name: '当前天气', description: '查询指定城市当前天气状况' },
      { id: 'get_forecast', name: '天气预报', description: '查询未来7天天气预报' },
    ],
  },
});

// 启动时注册到平台
sdk.start().catch(console.error);

// 进程退出时注销
process.on('SIGINT', () => sdk.stop().then(() => process.exit(0)));
process.on('SIGTERM', () => sdk.stop().then(() => process.exit(0)));

// ═══════════════════════════════════════════════════════════════
// 平台代理端点：接收平台转发来的消息（A2A 官方标准为 JSON-RPC POST / SendMessage）
// ═══════════════════════════════════════════════════════════════

/**
 * POST /sendMessage
 * 平台调用此端点，将其他 Agent 发来的消息送达
 */
app.post('/sendMessage', async (req: Request, res: Response) => {
  try {
    const message = req.body;
    // message 结构：
    // {
    //   fromAgentId: 'agent_001',
    //   toAgentId: 'agent_weather_001',
    //   taskId: 'task_xyz_789',
    //   messageType: 'TASK_REQUEST',
    //   payload: { body: '北京明天天气？', input: { city: '北京', date: '2025-05-10' } },
    //   _session_context: { goal: '...', userInput: '...', artifacts: [...] },
    //   _trace: { traceId: '...', spanId: '...', depth: 2 }
    // }

    console.log(`[Agent] Received message from ${message.fromAgentId}: ${message.payload.body}`);

    // 调用 LLM 处理请求
    const result = await handleMessage(message);

    // 返回结果（平台会转发回发送方）
    res.json(result);
  } catch (error) {
    console.error('[Agent] Error handling message:', error);
    res.status(500).json({ error: 'Internal agent error', message: (error as Error).message });
  }
});

/**
 * 消息处理核心逻辑
 */
async function handleMessage(message: any): Promise<{ body: string; artifact?: any }> {
  const { payload, _session_context, taskId } = message;

  // 构建给 LLM 的上下文
  const messages: any[] = [
    {
      role: 'system',
      content: `你是一个天气查询助手。当前任务：${_session_context?.goal || '协助用户查询天气'}。`,
    },
    {
      role: 'user',
      content: payload.body,
    },
  ];

  // 如果有历史 artifacts，注入上下文
  if (_session_context?.artifacts?.length > 0) {
    messages.push({
      role: 'system',
      content: `历史上下文：${JSON.stringify(_session_context.artifacts)}`,
    });
  }

  // 调用 LLM
  const completion = await openai.chat.completions.create({
    model: 'gpt-4',
    messages,
    tools: [sdk.getDiscoverTool(), sdk.getSendMessageTool()],  // 注入平台工具
  });

  const choice = completion.choices[0];

  // 如果 LLM 决定调用工具（如 discover_agents 或 send_message_to_agent）
  if (choice.message.tool_calls) {
    const toolCall = choice.message.tool_calls[0];
    const toolName = toolCall.function.name;
    const toolArgs = JSON.parse(toolCall.function.arguments);

    console.log(`[Agent] LLM decided to call tool: ${toolName}`, toolArgs);

    if (toolName === 'discover_agents') {
      const agents = await sdk.getDiscoverTool().execute(toolArgs);
      return {
        body: `我发现以下 Agent 可以协助：${JSON.stringify(agents.map((a: any) => a.name))}`,
        artifact: { toolResult: agents },
      };
    }

    if (toolName === 'send_message_to_agent') {
      const result = await sdk.getSendMessageTool().execute(toolArgs);
      return {
        body: `已委托其他 Agent 处理，结果：${JSON.stringify(result)}`,
        artifact: { toolResult: result },
      };
    }
  }

  // 正常文本回复
  return {
    body: choice.message.content || '处理完成',
  };
}

// ═══════════════════════════════════════════════════════════════
// 健康检查（供平台 / 负载均衡器调用）
// ═══════════════════════════════════════════════════════════════

app.get('/health', (_req: Request, res: Response) => {
  res.json({ status: 'ok', agentId: sdk.getAgentId(), timestamp: new Date().toISOString() });
});

// ═══════════════════════════════════════════════════════════════
// 启动
// ═══════════════════════════════════════════════════════════════

app.listen(PORT, () => {
  console.log(`[Agent] A2A Server listening on port ${PORT}`);
});

Agent Server 必须实现的端点

EndpointMethod说明
/sendMessagePOST必须：接收平台转发的同步消息，返回处理结果
/sendMessageStreamPOST可选：接收流式消息，返回 SSE 流（如果 Agent 支持 streaming）
/healthGET推荐：健康检查，返回 Agent 状态

重要：Agent 的 /sendMessage 只能被平台调用，不应该暴露给公网。生产环境中应配置防火墙或 mTLS，确保只有平台 IP 能访问。

// examples/flight-agent.ts
// ───────────────────────────────────────────────────────────────

import express, { Request, Response } from 'express';
import { A2AAgentSDK } from '@your-org/a2a-sdk';
import { OpenAI } from 'openai';

const app = express();
app.use(express.json());

const PLATFORM_URL = process.env.PLATFORM_URL || 'http://localhost:3000';
const AGENT_TOKEN = process.env.AGENT_TOKEN!;
const OPENAI_API_KEY = process.env.OPENAI_API_KEY!;
const PORT = 8005;

const openai = new OpenAI({ apiKey: OPENAI_API_KEY });

// ═══════════════════════════════════════════════════════════════
// 初始化 SDK
// ═══════════════════════════════════════════════════════════════

const sdk = new A2AAgentSDK({
  platformUrl: PLATFORM_URL,
  token: AGENT_TOKEN,
  agentCard: {
    name: 'flight-booking-agent',
    description: '机票查询与预订 Agent，支持搜索、比价、预订',

    version: '1.0.0',

    skills: [
      { id: 'search_flights', name: '机票搜索', description: '根据出发地、目的地、日期搜索航班' },
      { id: 'compare_prices', name: '机票比价', description: '比较不同航班的价格和舱位' },
      { id: 'book_flight', name: '机票预订', description: '预订指定航班的机票' },
    ],

sdk.start().catch(console.error);
process.on('SIGINT', () => sdk.stop().then(() => process.exit(0)));

// ═══════════════════════════════════════════════════════════════
// Mock 数据库（实际应接航司 API）
// ═══════════════════════════════════════════════════════════════

const MOCK_FLIGHTS: Record<string, any[]> = {
  '北京-成都-2025-05-14': [
    { flightNo: 'CA4102', airline: '国航', dep: '08:00', arr: '11:30', prices: { economy: 800, business: 1500, first: 3000 } },
    { flightNo: 'MU5401', airline: '东航', dep: '14:00', arr: '17:20', prices: { economy: 720, business: 1400, first: 2800 } },
    { flightNo: 'CZ3903', airline: '南航', dep: '19:00', arr: '22:10', prices: { economy: 650, business: 1300, first: 2600 } },
  ],
};

// ═══════════════════════════════════════════════════════════════
// 消息处理器
// ═══════════════════════════════════════════════════════════════

app.post('/sendMessage', async (req: Request, res: Response) => {
  try {
    const { payload, _session_context, taskId, fromAgentId } = req.body;
    console.log(`[FlightAgent] From ${fromAgentId}: ${payload.body}`);

    // 解析用户意图
    const intent = await parseIntent(payload.body, _session_context);

    let result: { body: string; artifact?: any };

    switch (intent.action) {
      case 'search':
        result = await searchFlights(intent.params);
        break;
      case 'compare':
        result = await compareFlights(intent.params);
        break;
      case 'book':
        result = await bookFlight(intent.params);
        break;
      default:
        result = { body: '我可以帮您搜索航班、比价或预订机票。请告诉我出发地、目的地和日期。' };
    }

    res.json(result);
  } catch (error) {
    console.error('[FlightAgent] Error:', error);
    res.status(500).json({ error: '机票查询失败', message: (error as Error).message });
  }
});

/**
 * 用 LLM 解析用户意图
 */
async function parseIntent(body: string, context: any): Promise<{ action: string; params: any }> {
  const completion = await openai.chat.completions.create({
    model: 'gpt-4',
    messages: [
      {
        role: 'system',
        content: `你是一个机票查询助手。解析用户意图，返回 JSON：
{ "action": "search|compare|book", "params": {...} }

当前上下文：${JSON.stringify(context?.sharedContext || {})}`,
      },
      { role: 'user', content: body },
    ],
    response_format: { type: 'json_object' },
  });

  return JSON.parse(completion.choices[0].message.content || '{}');
}

/**
 * 搜索航班
 */
async function searchFlights(params: any): Promise<{ body: string; artifact: any }> {
  const key = `${params.from}-${params.to}-${params.date}`;
  const flights = MOCK_FLIGHTS[key] || [];

  if (flights.length === 0) {
    return { body: `抱歉，${params.from}到${params.to}在${params.date}没有航班。`, artifact: { flights: [] } };
  }

  const options = flights.map((f: any) =>
    `${f.flightNo}(${f.airline}) ${f.dep}-${f.arr} 经济舱¥${f.prices.economy}/商务舱¥${f.prices.business}/头等舱¥${f.prices.first}`
  ).join('\n');

  return {
    body: `找到 ${flights.length} 个航班：\n${options}\n\n请告诉我您要选择哪个航班和舱位。`,
    artifact: { flights, query: params },
  };
}

/**
 * 比价（返回最便宜选项）
 */
async function compareFlights(params: any): Promise<{ body: string; artifact: any }> {
  const key = `${params.from}-${params.to}-${params.date}`;
  const flights = MOCK_FLIGHTS[key] || [];

  const cheapest = flights.reduce((min: any, f: any) =>
    f.prices.economy < min.prices.economy ? f : min
  , flights[0]);

  return {
    body: `最便宜的航班是 ${cheapest.flightNo}(${cheapest.airline})，经济舱 ¥${cheapest.prices.economy}。`,
    artifact: { cheapest, allOptions: flights },
  };
}

/**
 * 预订
 */
async function bookFlight(params: any): Promise<{ body: string; artifact: any }> {
  const { flightNo, cabin = 'economy' } = params;

  return {
    body: `已为您预订 ${flightNo} 的${cabin === 'economy' ? '经济舱' : cabin === 'business' ? '商务舱' : '头等舱'}机票。预订号：BK${Date.now()}。`,
    artifact: { bookingId: `BK${Date.now()}`, flightNo, cabin, status: 'confirmed' },
  };
}

// 健康检查
app.get('/health', (_req: Request, res: Response) => {
  res.json({ status: 'ok', agent: 'flight-booking-agent', agentId: sdk.getAgentId() });
});

app.listen(PORT, () => {
  console.log(`[FlightAgent] Running on port ${PORT}`);
});
